56int Player::getVal() const {
57 int total = 0;
58 int aces = 0;
59
60 // Calculate non-ace values first
61 for (const auto& card : hand) {
62 int val = card.getValue();
63 if (val > 10) val = 10;
64 if (val == 1) {
65 aces++;
66 } else {
67 total += val;
68 }
69 }
70
71 // Handle aces
72 for (int i = 0; i < aces; ++i) {
73 if (total + 11 <= 21) {
74 total += 11;
75 } else {
76 total += 1;
77 }
78 }
79
80 return total;
81}
